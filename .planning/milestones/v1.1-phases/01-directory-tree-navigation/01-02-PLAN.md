Create new FolderTreeView.swift with a recursive folder tree for the sidebar:

```swift
struct FolderTreeView: View {
    @ObservedObject var vm: DatasetViewModel

    var body: some View {
        List(selection: $vm.selectedFolderID) {
            // Show root folder as a header section
            if let rootURL = vm.directoryURL {
                Section(rootURL.lastPathComponent) {
                    ForEach(vm.folderTree) { node in
                        FolderRow(node: node, vm: vm)
                    }
                }
            }
        }
        .listStyle(.sidebar)
    }
}

struct FolderRow: View {
    let node: FileNode
    @ObservedObject var vm: DatasetViewModel

    var body: some View {
        if let children = node.children, !children.isEmpty {
            DisclosureGroup {
                ForEach(children) { child in
                    FolderRow(node: child, vm: vm)
                }
            } label: {
                Label(node.name, systemImage: node.icon)
                    .tag(node.id)
            }
        } else if node.children?.isEmpty == true {
            // Empty folder - show without disclosure
            Label(node.name, systemImage: "folder")
                .tag(node.id)
        } else {
            // children == nil means not loaded yet
            // Show with disclosure that loads on expand
            DisclosureGroup {
                // Will be populated when expanded
            } label: {
                Label(node.name, systemImage: node.icon)
                    .tag(node.id)
            }
            .onAppear {
                // Could trigger lazy load here if needed
            }
        }
    }
}
```

Key patterns:
- Use DisclosureGroup for expandable folders (NOT OutlineGroup - buggy on macOS)
- Recursive FolderRow for nested folders
- Use .tag(node.id) for selection binding
- SF Symbols: folder, folder.fill for folders
